def play_game(model, qmaze, pirate_cell, max_steps=None):
    qmaze.reset(pirate_cell)
    envstate = qmaze.observe()
    steps = 0
    if max_steps is None:
        max_steps = qmaze.maze.size * 4  # safety cutoff

    while steps < max_steps:
        state = np.asarray(envstate, dtype=np.float32)
        if state.ndim == 1:
            state = np.expand_dims(state, axis=0)

        q_values = model(state, training=False).numpy()
        action = np.argmax(q_values[0])

        envstate, reward, game_status = qmaze.act(action)
        steps += 1

        if game_status == 'win':
            return True
        elif game_status == 'lose':
            return False

    return False  # timed out with no result

def completion_check(model, maze_or_qmaze, max_steps=None):
    # Accept either raw numpy maze or TreasureMaze instance
    if isinstance(maze_or_qmaze, TreasureMaze):
        qmaze = maze_or_qmaze
    else:
        qmaze = TreasureMaze(maze_or_qmaze)

    for cell in qmaze.free_cells:
        if not qmaze.valid_actions(cell):
            continue
        if not play_game(model, qmaze, cell, max_steps=max_steps):
            return False
    return True

@tf.function
def train_step(x, y):
    with tf.GradientTape() as tape:
        q_values = model(x, training=True)
        loss = loss_fn(y, q_values)
    grads = tape.gradient(loss, model.trainable_variables)
    optimizer.apply_gradients(zip(grads, model.trainable_variables))
    return loss

TARGET_FREQ = 50

def qtrain(model, maze, **opt):
    """
    Trains a Deep Q-Learning agent to navigate the TreasureMaze.

    This implementation uses a Target Network for training stability and 
    aggressive reward shaping for fast convergence.
    """
    global epsilon  # exploration factor (should be initialized outside this function)

    # Training Hyperparameters
    n_epoch = opt.get('n_epoch', 1000)
    max_memory = opt.get('max_memory', 1000)
    data_size = opt.get('data_size', 64) 
    target_update_freq = opt.get('target_update_freq', TARGET_FREQ)
    
    start_time = datetime.datetime.now()

    # Create the Target Model
    target_model = clone_model(model)
    target_model.set_weights(model.get_weights())
    
    # Initialize Experience Replay object with BOTH models.
    # GameExperience handles the stable Q-target calculation using target_model.
    experience = GameExperience(model, target_model, max_memory=max_memory)
    target_update_counter = 0

    # Environment setup
    qmaze = TreasureMaze(maze)
    win_history = []
    hsize = qmaze.maze.size // 2 
    win_rate = 0.0

    # Training Loop
    for epoch in range(n_epoch):
        
        game_over = False
        loss = 0.0
        n_episodes = 0
        
        # PSEUDOCODE: Reset the environment at a random starting cell
        agent_cell = random.choice(qmaze.free_cells)
        qmaze.reset(agent_cell)
        
        # PSEUDOCODE: Set the initial environment state
        envstate = qmaze.observe()
        
        # While game status is not game over:
        while not game_over:
            
            valid_actions = qmaze.valid_actions()
            if not valid_actions:
                break 
                
            prev_envstate = envstate
            
            # PSEUDOCODE: Decide on an action (Exploration vs. Exploitation)
            if np.random.rand() < epsilon:
                action = random.choice(valid_actions) # Exploration
            else:
                q_values = experience.predict(envstate)
                
                max_q = np.max(q_values)
                
                # Find all actions that match the maximum Q-value
                best_actions = np.where(q_values >= max_q - 1e-4)[0]
                
                exploitable_actions = [a for a in best_actions if a in valid_actions]

                if not exploitable_actions:
                    # Fallback to pure exploration if the 'best' action isn't valid
                    action = random.choice(valid_actions)
                else:
                    # Choose randomly from the best valid options
                    action = random.choice(exploitable_actions)
                
            # PSEUDOCODE: Retrieve the values below from the act() method.
            # env_state, reward_base, game_status = qmaze.act(action)
            envstate, reward_base, game_status = qmaze.act(action)
            
            # Aggressive Reward Shaping (Overrides mild reward_base for fast learning)
            if game_status == 'win':
                reward = 50.0   # Massive positive reward
            elif game_status == 'lose':
                reward = -5.0 
            else:
                reward = reward_base
                
            # Check if the episode has ended
            if game_status == 'win' or game_status == 'lose':
                game_over = True
                
            # PSEUDOCODE: Store the episode in the Experience replay object
            episode = [prev_envstate, action, reward, envstate, game_over]
            experience.remember(episode)
            
            n_episodes += 1
        
        # PSEUDOCODE: Track the wins and losses from the game_status
        win_history.append(1 if game_status == 'win' else 0)
        
        # Maintain history window size
        if len(win_history) > hsize:
            del win_history[0]

        if len(win_history) >= hsize:
             win_rate = sum(win_history[-hsize:]) / hsize
        
        # Epsilon Decay (Controlled linear decay for fast finish)
        end_decay_epoch = 500 

        if epoch < end_decay_epoch:
            epsilon = 1.0 - (1.0 - epsilon_min) * (epoch / end_decay_epoch)
        else:
            epsilon = epsilon_min

        if win_rate > 0.9: 
             epsilon = epsilon_min
        
        # PSEUDOCODE: Train neural network model and evaluate loss
        if len(experience.memory) >= data_size:
            
            # Get data batch. GameExperience.get_data uses the target_model for stability.
            inputs, targets = experience.get_data(batch_size=data_size)
            
            # Train model (Main Model)
            history = model.fit(
                inputs, 
                targets, 
                epochs=1, 
                batch_size=64,
                verbose=0
            )
            
            # Evaluate loss
            loss = model.evaluate(
                inputs, 
                targets, 
                verbose=0
            )

            # Update Target Network Weights
            target_update_counter += 1
            if target_update_counter >= target_update_freq:
                target_model.set_weights(model.get_weights())
                target_update_counter = 0

    # Print the epoch, loss, episode, win count, win rate, and time for each epoch
        dt = datetime.datetime.now() - start_time
        t = format_time(dt.total_seconds())
        template = "Epoch: {:03d}/{:d} | Loss: {:.4f} | Episodes: {:d} | Win count: {:d} | Win rate: {:.3f} | time: {}"
        print(template.format(epoch, n_epoch-1, loss, n_episodes, sum(win_history), win_rate, t))
        
        # PSEUDOCODE: If the win rate is above the threshold and your model passes the completion check...
        if win_rate >= 0.99 and completion_check(model, qmaze):
            print("Reached 100%% win rate at epoch: %d" % (epoch,))
            break
        
    # Determine the total time for training
    dt = datetime.datetime.now() - start_time
    seconds = dt.total_seconds()
    t = format_time(seconds)

    print("n_epoch: %d, max_mem: %d, data: %d, time: %s" % (epoch, max_memory, data_size, t))
    return seconds

def format_time(seconds):
    if seconds < 400:
        s = float(seconds)
        return "%.1f seconds" % (s,)
    elif seconds < 4000:
        m = seconds / 60.0
        return "%.2f minutes" % (m,)
    else:
        h = seconds / 3600.0
        return "%.2f hours" % (h,)
